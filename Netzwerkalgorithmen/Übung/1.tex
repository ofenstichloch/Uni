void dfs(const graph& G, node u, node_array<int>& dfsnum, compnum, node_array<edge>& current_edge, int& dfs_count, int& comp_count)
stack<node> S
S.push(u)
dfsnum[u] = ++dfs_count
while(!S.empty()) {
	node v = s.top()
	edge e = current_edge[v]
	if (e == NULL){
		s.pop()
		compnum[v] = ++comp_count	
	}else{
		node w = G.target(e)
		if(dfsnum[w]==0){
			s.push(w)	
			dfsnum[w] = ++dfsnum
			current_edge[w] = G.out_succ(e)
		}
	}
}
void DFSNUM(const graph& G, node_array<int>& dfsnum, compnum)
int dfs_count = 0
int comp_count = 0
node_array<edge> current_edge(G)
node v
forall_nodes(v,G){
	dfsnum[v] = 0
	current_edge[v] = G.first_out_edge(v)
}
forall_nodes(v,G){
	if(dfsnum[v]==0){
		dfs(G,v,dfsnum,compnum,current_edge,dfs_count,comp_count)
	}
}
\end{lstlisting}
